for(trait in n_studies_trait$MAPPED_TRAIT[1:1000]){
  
  euro_n = gwas_ancest_info |>
           filter(ancestry_group == "European") |>
           filter(MAPPED_TRAIT == trait) |>
           pull(NUMBER_OF_INDIVIDUALS) |>
           sum(na.rm = T)
  
  total_n = gwas_ancest_info |>
            filter(MAPPED_TRAIT == trait) |>
            pull(NUMBER_OF_INDIVIDUALS) |>
            sum(na.rm = T)
